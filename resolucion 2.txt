// Definici칩n de la funci칩n que agrega la propiedad 'question'
function addQuestionProperty(inputArray) {
  // Array que almacenar치 los desarrolladores con detalles faltantes
  const developersWithMissingDetails = [];

  // Iterar sobre cada desarrollador en el array de entrada
  inputArray.forEach(developer => {
    // Obtener las propiedades con valores nulos
    const missingDetails = Object.keys(developer).filter(key => developer[key] === null);

    // Si hay detalles faltantes, agregar la propiedad 'question'
    if (missingDetails.length > 0) {
      // Crear el mensaje de la pregunta
      developer.question = `Hi, could you please provide your ${missingDetails.join(' or ')}?`;

      // Agregar el desarrollador al array de resultados
      developersWithMissingDetails.push(developer);
    }
  });

  // Retornar el array con los desarrolladores con detalles faltantes
  return developersWithMissingDetails;
}

// Ejemplo de datos de entrada
let list1 = [
  { firstName: null, lastName: 'I.', country: 'Argentina', continent: 'Americas', age: 35, language: 'Java' },
  { firstName: 'Lukas', lastName: 'X.', country: 'Croatia', continent: 'Europe', age: 35, language: null },
  { firstName: 'Madison', lastName: 'U.', country: 'United States', continent: 'Americas', age: 32, language: 'Ruby' }
];

// Llamada a la funci칩n con el ejemplo de datos
let result = addQuestionProperty(list1);
